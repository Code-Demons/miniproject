        function drawSkier(skier) {
            const skierImage = new Image();
            skierImage.src = skier.imageUrl;
            ctx.drawImage(skierImage, skier.position, 70 * skier.lane + 63, 50, 50); // Adjust width and height as needed
        }
        function updateSkiRace() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            let allSkiersDone = true;
            for (const skier of skiers) {
                if (!skier.done) {
                    skier.position += skier.speed;
                    if (skier.position > canvas.width - 20) {
                        skier.position = canvas.width - 20;
                        skier.done = true;
                    } else {
                        allSkiersDone = false;
                    }
                }
                drawSkier(skier);
            }
            if (!allSkiersDone) {
                requestAnimationFrame(updateSkiRace);
            }
        }
        function generatePartiallyOrganizedArray() {
            const arrayLength = Math.floor(Math.random() * 10) + 5; // Random length between 5 and 14
            const partiallyOrganizedArray = Array.from({ length: arrayLength }, (_, index) => index + 1);
            // Shuffle the array to introduce some randomness
            for (let i = partiallyOrganizedArray.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [partiallyOrganizedArray[i], partiallyOrganizedArray[j]] = [partiallyOrganizedArray[j], partiallyOrganizedArray[i]];
            }
            return partiallyOrganizedArray;
        }
        function runSkiRace() {
            if (animationFrameId) {
                cancelAnimationFrame(animationFrameId);

            resetSkiRace();
            // Generate partially organized arrays for each skier
            const partiallyOrganizedArrays = skiers.map(() => generatePartiallyOrganizedArray());
            const endpoints = partiallyOrganizedArrays.map((array, index) =>
                `http://localhost:8085/sort/${skiers[index].name.toLowerCase()}/${array}`
            );
            let maxTime = 0;
            let results = [];
            for (let i = 0; i < skiers.length; i++) {
                // Simulate asynchronous fetch with a delay
                const response = await simulateFetch(endpoints[i]);
                const data = await response.json();
                const timeTaken = data.timeTaken;
                maxTime = Math.max(maxTime, timeTaken);
                results.push({ name: skiers[i].name, sortedArray: data.sortedArray, timeTaken: timeTaken });
            }
            for (let i = 0; i < skiers.length; i++) {
                skiers[i].speed = calculateSpeed(results[i].timeTaken, maxTime);
            }
            displayResults(results);
            startSkiRace();
        }
